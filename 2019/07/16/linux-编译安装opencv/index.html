linux 编译安装opencv

Linux 编译OpenCV

安装

  1. 安装类库
    sudo apt-get install cmake git libgtk2.0-dev pkg-config libavcodec-dev libavformat-dev libswscale-dev

sudo apt-get install python-dev python-numpy libtbb2 libtbb-dev libjpeg-dev libpng-dev libtiff-dev libjasper-dev libdc1394-22-dev

  1. cmake

cmake -DCMAKE_BUILD_TYPE=RELEASE -DCMAKE_INSTALL_PREFIX=/usr/local -DOPENCV_GENERATE_PKGCONFIG=ON ..

最后一个参数会生成 /usr/local/lib/pkgconfig

  1. 编译
    make -j8

  2. 安装
    make install

pkgconfig 运行

默认情况,pkg-config首先在 /usr/lib/pkgconfig/ /usr/local/lib/pkgconfig 查找xxx.pc
没有的话在 PKG_CONFIG_PATH

pkg-config –libs –cflags opencv
查看gcc的CFLAGS参数

Linux 变量

C_INCLUDE_PATH C 头文件
CPLUS_INCLUDE_PATH C++ 头文件
LD_LIBRARY_PATH 动态链接库路径
LIBRARY_PATH 静态库的路径

运行opencv

  1. 直接指定 -I —L -l
    g++ test.cpp -I /usr/local/include/opencv4 -lopencv_features2d -lopencv_calib3d -opencv_imgproc -lopencv_core -lopencv_highgui -lopencv_imgcodecs -o test

缺少lib可以用下面的方法
nm -A lib64/libopencv*.a |grep “头文件”

  1. pkg-config
    pkg-config –help
    pkg-config –modversion opencv4
    g++ test.cpp pkg-config --libs --cflags opencv4
    生成a.out

  2. cmake CMakeList.txt
    cmake_minimum_required(VERSION 2.8)
    project(opencv)
    set(OpneCV_DIR /path/to/opencv-x86/)
    find_package(OpenCV REQUIRED)
    add_executable(opencv try.cpp)
    target_link_libraries(opencv ${OpenCV_LIBS})

  3. 动态链接库
    /etc/ld.so.conf
    include /etc/ld.so.conf.d/*.conf

sudo ldconfig
